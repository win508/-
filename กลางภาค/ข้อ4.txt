min_num = None
while True:
    try:
        num = float(input("ระบุจำนวนจริงบวก: "))
        if num <= 0:
            break
        if min_num is None or num < min_num:
            min_num = num
    except:
        print("กรุณากรอกตัวเลขให้ถูกต้อง")
        break

if min_num is not None:
    print("จำนวนที่น้อยที่สุดเท่ากับ", min_num)
else:
    print("ไม่มีจำนวนจริงบวก")
